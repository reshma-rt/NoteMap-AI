import React, { useState, useRef, useEffect } from 'react';
import { Home, Upload, LayoutDashboard, Sun, Moon, FileText, Search, Download, ExternalLink, ChevronDown, ChevronRight, X, CheckCircle, Clock, AlertCircle, Eye, Settings, BookOpen, Tag, MapPin, Sparkles } from 'lucide-react';

export default function NoteMap() {
  const [currentPage, setCurrentPage] = useState('home');
  const [isDark, setIsDark] = useState(true);
  const [selectedSubject, setSelectedSubject] = useState('All Subjects');
  const [searchQuery, setSearchQuery] = useState('');
  const [uploadedFiles, setUploadedFiles] = useState([]);
  const [isProcessing, setIsProcessing] = useState(false);
  const [dragActive, setDragActive] = useState(false);
  const [previewFile, setPreviewFile] = useState(null);
  const [showPreview, setShowPreview] = useState(false);
  const [processedNotes, setProcessedNotes] = useState([]);
  const [apiEndpoint, setApiEndpoint] = useState('http://localhost:5000/api/upload');
  const [showSettings, setShowSettings] = useState(false);
  const [backendInfo, setBackendInfo] = useState(null);
  const [expandedTOC, setExpandedTOC] = useState({});
  const [selectedFormat, setSelectedFormat] = useState('md');
  const [downloadStatus, setDownloadStatus] = useState({});
  const [tocSearchQuery, setTocSearchQuery] = useState('');
  const [selectedTOCEntry, setSelectedTOCEntry] = useState(null);
  const fileInputRef = useRef(null);

  // Check backend health on mount
  useEffect(() => {
    checkBackendHealth();
  }, []);

  const checkBackendHealth = async () => {
    try {
      const res = await fetch('http://localhost:5000/api/health');
      const data = await res.json();
      setBackendInfo(data);
    } catch (err) {
      console.error('Backend not available:', err);
    }
  };

  // Handle file upload
  const handleFileUpload = async (files) => {
    const validFiles = Array.from(files).filter(file => {
      const validTypes = ['image/png', 'image/jpeg', 'image/jpg', 'application/pdf'];
      const maxSize = 10 * 1024 * 1024;
      
      if (!validTypes.includes(file.type)) {
        alert(`${file.name} is not a valid file type.`);
        return false;
      }
      if (file.size > maxSize) {
        alert(`${file.name} is too large. Max 10MB.`);
        return false;
      }
      return true;
    });

    if (validFiles.length === 0) return;

    setIsProcessing(true);
    
    for (const file of validFiles) {
      const reader = new FileReader();
      
      reader.onload = async (e) => {
        const newFile = {
          id: Date.now() + Math.random(),
          name: file.name,
          size: (file.size / 1024).toFixed(2) + ' KB',
          type: file.type,
          status: 'processing',
          progress: 0,
          previewUrl: e.target.result,
          file: file
        };
        
        setUploadedFiles(prev => [...prev, newFile]);
        
        // Progress simulation
        let progress = 0;
        const progressInterval = setInterval(() => {
          progress += 15;
          if (progress <= 90) {
            setUploadedFiles(prev => prev.map(f => 
              f.id === newFile.id ? { ...f, progress } : f
            ));
          }
        }, 400);
        
        // Process file
        try {
          const result = await extractTextFromFile(file);
          clearInterval(progressInterval);
          
          if (!result || !result.success) {
            throw new Error(result?.error || 'Processing failed');
          }

          // Update file status
          setUploadedFiles(prev => prev.map(f =>
            f.id === newFile.id ? {
              ...f,
              status: 'completed',
              progress: 100,
              ...result
            } : f
          ));

          // Add to processed notes
          const processedNote = {
            id: Date.now() + Math.random(),
            filename: file.name,
            subject: result.subject || 'Other',
            date: result.date || new Date().toLocaleDateString('en-US', { month: 'short', day: 'numeric' }),
            charCount: result.charCount || 0,
            wordCount: result.wordCount || 0,
            extractedText: result.extractedText || result.text || '',
            sections: result.sections || [],
            concepts: result.concepts || [],
            tableOfContents: result.tableOfContents || null,
            processingInfo: result.processingInfo || null,
            totalSections: (result.sections || []).length,
            preview: (result.preview || result.extractedText || '').substring(0, 150) + '...'
          };

          setProcessedNotes(prev => [...prev, processedNote]);
          
        } catch (error) {
          clearInterval(progressInterval);
          console.error('Processing error:', error);
          
          setUploadedFiles(prev => prev.map(f => 
            f.id === newFile.id ? { 
              ...f, 
              status: 'error', 
              progress: 0,
              errorMessage: error.message 
            } : f
          ));
          
          alert(`Failed to process ${file.name}: ${error.message}`);
        }
      };
      
      reader.readAsDataURL(file);
    }
    
    setIsProcessing(false);
  };

  // Extract text using backend API
  const extractTextFromFile = async (file) => {
    const formData = new FormData();
    formData.append("file", file);

    const endpoints = [
      apiEndpoint,
      "http://localhost:5000/api/upload",
      "http://127.0.0.1:5000/api/upload"
    ];

    const uniqueEndpoints = [...new Set(endpoints)];
    let lastError = null;
    
    for (const endpoint of uniqueEndpoints) {
      try {
        const res = await fetch(endpoint, {
          method: "POST",
          body: formData,
        });

        if (!res.ok) {
          const errorText = await res.text();
          lastError = new Error(`API error ${res.status}: ${errorText}`);
          continue;
        }

        const data = await res.json();
        return data;
        
      } catch (err) {
        lastError = err;
        continue;
      }
    }

    throw lastError || new Error("Could not connect to backend");
  };

  // Preview handlers
  const handlePreview = (file) => {
    setPreviewFile(file);
    setShowPreview(true);
  };

  const closePreview = () => {
    setShowPreview(false);
    setPreviewFile(null);
  };

  // Download functions
  const downloadNotesBySubject = async (subject) => {
    const notes = subject === 'All Subjects' 
      ? processedNotes 
      : processedNotes.filter(n => n.subject === subject);
    
    if (notes.length === 0) {
      alert('No notes found for this subject');
      return;
    }

    let content = generateCombinedMarkdown(notes, subject);

    const blob = new Blob([content], { type: 'text/plain' });
    const url = URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.href = url;
    link.download = `${subject.replace(/\s+/g, '_')}_All_Notes.txt`;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    URL.revokeObjectURL(url);
    
    alert(`Downloaded ${notes.length} notes from ${subject}!`);
  };

  const downloadNote = async (format, note) => {
    setDownloadStatus(prev => ({ ...prev, [note.id]: 'Preparing download...' }));

    setTimeout(() => {
      let content = '';
      let filename = '';
      let mimeType = '';

      if (format === 'txt') {
        content = generateTextContent(note);
        filename = `${note.filename.replace(/\.[^/.]+$/, '')}_notes.txt`;
        mimeType = 'text/plain';
      } else if (format === 'md') {
        content = generateMarkdownContent(note);
        filename = `${note.filename.replace(/\.[^/.]+$/, '')}_notes.md`;
        mimeType = 'text/markdown';
      } else if (format === 'json') {
        content = JSON.stringify(note, null, 2);
        filename = `${note.filename.replace(/\.[^/.]+$/, '')}_notes.json`;
        mimeType = 'application/json';
      }

      const blob = new Blob([content], { type: mimeType });
      const url = URL.createObjectURL(blob);
      const link = document.createElement('a');
      link.href = url;
      link.download = filename;
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
      URL.revokeObjectURL(url);

      setDownloadStatus(prev => ({ ...prev, [note.id]: `‚úì Downloaded ${filename}` }));
      setTimeout(() => {
        setDownloadStatus(prev => {
          const newStatus = { ...prev };
          delete newStatus[note.id];
          return newStatus;
        });
      }, 3000);
    }, 500);
  };

  const generateCombinedMarkdown = (notes, subject) => {
    let content = [];
    content.push(`# ${subject} - Complete Notes Collection\n`);
    content.push(`**Generated:** ${new Date().toLocaleString()}\n`);
    content.push(`**Total Documents:** ${notes.length}\n`);
    content.push('---\n');

    notes.forEach((note, idx) => {
      content.push(`\n## Document ${idx + 1}: ${note.filename}\n`);
      content.push(`**Subject:** ${note.subject} | **Date:** ${note.date}\n`);
      
      if (note.tableOfContents) {
        content.push('\n### üìë Table of Contents\n');
        note.tableOfContents.entries.forEach(entry => {
          const indent = '  '.repeat(entry.level - 1);
          content.push(`${indent}- **${entry.title}** (Page ${entry.page})`);
        });
        content.push('');
      }

      content.push('\n### üìù Content\n');
      note.sections.forEach(section => {
        const heading = '#'.repeat((section.level || 1) + 3);
        content.push(`\n${heading} ${section.title}`);
        if (section.topic) {
          content.push(`\n> **Topic:** ${section.topic}`);
        }
        if (section.page) {
          content.push(`> **Page:** ${section.page}\n`);
        }
        content.push(`${section.desc}\n`);
      });

      if (note.concepts && note.concepts.length > 0) {
        content.push('\n### üîë Key Concepts\n');
        note.concepts.forEach(concept => {
          content.push(`- \`${concept}\``);
        });
      }

      content.push('\n---\n');
    });

    return content.join('\n');
  };

  const generateTextContent = (note) => {
    let content = [];
    content.push('='.repeat(80));
    content.push('ORGANIZED NOTES - NoteMap');
    content.push(`Generated: ${new Date().toLocaleString()}`);
    content.push('='.repeat(80));
    content.push('');
    content.push(`FILE: ${note.filename}`);
    content.push(`SUBJECT: ${note.subject}`);
    content.push(`DATE: ${note.date}`);
    content.push(`SECTIONS: ${note.totalSections}`);
    content.push('='.repeat(80));
    content.push('');

    if (note.tableOfContents) {
      content.push('TABLE OF CONTENTS:');
      content.push('-'.repeat(80));
      note.tableOfContents.entries.forEach(entry => {
        const indent = '  '.repeat(entry.level - 1);
        content.push(`${indent}${entry.id}. ${entry.title} (Page ${entry.page}, ${entry.topic})`);
      });
      content.push('');
    }

    content.push('CONTENT:');
    content.push('-'.repeat(80));
    note.sections.forEach((section, idx) => {
      content.push(`\n${idx + 1}. ${section.title}`);
      if (section.topic) {
        content.push(`   Topic: ${section.topic}`);
      }
      if (section.page) {
        content.push(`   Page: ${section.page} | Level: H${section.level}`);
      }
      content.push(`   ${section.desc}`);
      content.push('');
    });

    if (note.concepts && note.concepts.length > 0) {
      content.push('KEY CONCEPTS:');
      content.push('-'.repeat(80));
      content.push(note.concepts.join(', '));
      content.push('');
    }

    return content.join('\n');
  };

  const generateMarkdownContent = (note) => {
    let content = [];
    content.push('# Organized Notes - NoteMap\n');
    content.push(`**Generated:** ${new Date().toLocaleString()}\n`);
    content.push('---\n');
    content.push(`## üìÑ ${note.filename}\n`);
    content.push(`**Subject:** ${note.subject}  `);
    content.push(`**Date:** ${note.date}  `);
    content.push(`**Sections:** ${note.totalSections}\n`);

    if (note.tableOfContents) {
      content.push('### üìë Table of Contents\n');
      note.tableOfContents.entries.forEach(entry => {
        const indent = '  '.repeat(entry.level - 1);
        content.push(`${indent}- **${entry.title}** _(Page ${entry.page}, ${entry.topic})_`);
      });
      content.push('');
    }

    content.push('### üìù Content\n');
    note.sections.forEach(section => {
      const heading = '#'.repeat((section.level || 1) + 3);
      content.push(`\n${heading} ${section.title}`);
      if (section.topic) {
        content.push(`\n> **Topic:** ${section.topic}`);
      }
      if (section.page) {
        content.push(`> **Page:** ${section.page}\n`);
      }
      content.push(`${section.desc}\n`);
    });

    if (note.concepts && note.concepts.length > 0) {
      content.push('### üîë Key Concepts\n');
      note.concepts.forEach(concept => {
        content.push(`- \`${concept}\``);
      });
      content.push('');
    }

    return content.join('\n');
  };

  // Utility functions
  const getTopicColor = (topic) => {
    const colors = {
      'Statistics': 'bg-purple-500/10 text-purple-400 border-purple-500/30',
      'Data Structures': 'bg-blue-500/10 text-blue-400 border-blue-500/30',
      'Operating Systems': 'bg-green-500/10 text-green-400 border-green-500/30',
      'Computer Networks': 'bg-cyan-500/10 text-cyan-400 border-cyan-500/30',
      'Software Engineering': 'bg-orange-500/10 text-orange-400 border-orange-500/30',
      'Mathematics': 'bg-indigo-500/10 text-indigo-400 border-indigo-500/30',
      'Management': 'bg-pink-500/10 text-pink-400 border-pink-500/30',
      'Computer Science': 'bg-blue-500/10 text-blue-400 border-blue-500/30',
      'Cybersecurity': 'bg-red-500/10 text-red-400 border-red-500/30',
      'General': 'bg-gray-500/10 text-gray-400 border-gray-500/30',
      'Other': 'bg-gray-500/10 text-gray-400 border-gray-500/30'
    };
    return colors[topic] || colors['Other'];
  };

  const toggleTOCSection = (section) => {
    setExpandedTOC(prev => ({
      ...prev,
      [section]: !prev[section]
    }));
  };

  const searchTOC = (entries, query) => {
    if (!query.trim()) return entries;
    const lowerQuery = query.toLowerCase();
    return entries.filter(entry =>
      entry.title.toLowerCase().includes(lowerQuery) ||
      entry.topic.toLowerCase().includes(lowerQuery) ||
      entry.page.toString().includes(lowerQuery)
    );
  };

  const scrollToTOCEntry = (entry) => {
    setSelectedTOCEntry(entry);
  };

  const exportTOC = (note, format) => {
    let content = '';
    let filename = '';
    let mimeType = '';

    if (format === 'md') {
      content = generateTOCContent(note, 'md');
      filename = `${note.filename.replace(/\.[^/.]+$/, '')}_toc.md`;
      mimeType = 'text/markdown';
    } else if (format === 'txt') {
      content = generateTOCContent(note, 'txt');
      filename = `${note.filename.replace(/\.[^/.]+$/, '')}_toc.txt`;
      mimeType = 'text/plain';
    } else if (format === 'json') {
      content = JSON.stringify(note.tableOfContents, null, 2);
      filename = `${note.filename.replace(/\.[^/.]+$/, '')}_toc.json`;
      mimeType = 'application/json';
    }

    const blob = new Blob([content], { type: mimeType });
    const url = URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.href = url;
    link.download = filename;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    URL.revokeObjectURL(url);

    setDownloadStatus(prev => ({ ...prev, [note.id]: `‚úì TOC exported as ${filename}` }));
    setTimeout(() => {
      setDownloadStatus(prev => {
        const newStatus = { ...prev };
        delete newStatus[note.id];
        return newStatus;
      });
    }, 3000);
  };

  const generateTOCContent = (note, format) => {
    if (format === 'md') {
      let content = `# Table of Contents - ${note.filename}\n\n`;
      content += `**Generated:** ${new Date().toLocaleString()}\n\n`;
      content += '---\n\n';

      note.tableOfContents.entries.forEach(entry => {
        const indent = '  '.repeat(entry.level - 1);
        content += `${indent}- **${entry.title}** (Page ${entry.page})\n`;
        content += `${indent}  - Topic: ${entry.topic}\n\n`;
      });

      return content;
    } else if (format === 'txt') {
      let content = `TABLE OF CONTENTS - ${note.filename}\n`;
      content += `Generated: ${new Date().toLocaleString()}\n\n`;
      content += '='.repeat(50) + '\n\n';

      note.tableOfContents.entries.forEach(entry => {
        const indent = '  '.repeat(entry.level - 1);
        content += `${indent}${entry.id}. ${entry.title}\n`;
        content += `${indent}   Page: ${entry.page} | Topic: ${entry.topic}\n\n`;
      });

      return content;
    }
  };

  const removeFile = (fileId) => {
    setUploadedFiles(prev => prev.filter(f => f.id !== fileId));
  };

  // Drag and drop handlers
  const handleDrag = (e) => {
    e.preventDefault();
    e.stopPropagation();
    if (e.type === "dragenter" || e.type === "dragover") {
      setDragActive(true);
    } else if (e.type === "dragleave") {
      setDragActive(false);
    }
  };

  const handleDrop = (e) => {
    e.preventDefault();
    e.stopPropagation();
    setDragActive(false);
    
    if (e.dataTransfer.files && e.dataTransfer.files.length > 0) {
      handleFileUpload(e.dataTransfer.files);
    }
  };

  const handleFileInputChange = (e) => {
    if (e.target.files && e.target.files.length > 0) {
      handleFileUpload(e.target.files);
    }
  };

  // Get filtered notes
  const getFilteredNotes = () => {
    let filtered = processedNotes;

    if (selectedSubject !== 'All Subjects') {
      filtered = filtered.filter(note => note.subject === selectedSubject);
    }

    if (searchQuery.trim()) {
      const query = searchQuery.toLowerCase();
      filtered = filtered.filter(note => 
        note.filename.toLowerCase().includes(query) ||
        note.subject.toLowerCase().includes(query) ||
        note.preview.toLowerCase().includes(query) ||
        (note.sections && note.sections.some(section => 
          section.title.toLowerCase().includes(query) ||
          section.desc.toLowerCase().includes(query)
        )) ||
        (note.concepts && note.concepts.some(concept => 
          concept.toLowerCase().includes(query)
        ))
      );
    }

    return filtered;
  };

  // Get available subjects from processed notes
  const getAvailableSubjects = () => {
    const subjects = new Set(processedNotes.map(n => n.subject));
    return ['All Subjects', ...Array.from(subjects).sort()];
  };

  // Page renderers
  const renderHomePage = () => (
    <div className="min-h-screen flex flex-col items-center justify-center px-6">
      <div className={`mb-8 px-6 py-2 rounded-full flex items-center gap-2 ${
        isDark 
          ? 'bg-purple-900/40 border border-purple-500/30' 
          : 'bg-purple-100 border border-purple-300'
      }`}>
        <Sparkles size={18} className={isDark ? 'text-purple-300' : 'text-purple-600'} />
        <span className={`font-medium ${isDark ? 'text-purple-200' : 'text-purple-700'}`}>
          AI-Powered Note Organization
        </span>
      </div>
      
      {backendInfo && backendInfo.status === 'healthy' && (
        <div className={`mb-6 px-4 py-3 rounded-lg ${
          isDark ? 'bg-gray-800/50' : 'bg-white shadow-sm'
        }`}>
          <div className="flex flex-wrap justify-center gap-2">
            <span className={`text-xs px-3 py-1 rounded-full ${
              isDark ? 'bg-blue-900/30 text-blue-300' : 'bg-blue-100 text-blue-700'
            }`}>
              üîç Tesseract OCR
            </span>
            <span className={`text-xs px-3 py-1 rounded-full ${
              isDark ? 'bg-purple-900/30 text-purple-300' : 'bg-purple-100 text-purple-700'
            }`}>
              üìö Smart TOC Generation
            </span>
            <span className={`text-xs px-3 py-1 rounded-full ${
              isDark ? 'bg-indigo-900/30 text-indigo-300' : 'bg-indigo-100 text-indigo-700'
            }`}>
              üéØ Auto Classification
            </span>
          </div>
        </div>
      )}
      
      <div className="relative mb-6">
        <div className={`absolute inset-0 blur-3xl ${
          isDark 
            ? 'bg-gradient-to-r from-purple-600/30 via-blue-600/30 to-purple-600/30' 
            : 'bg-gradient-to-r from-purple-400/20 via-blue-400/20 to-purple-400/20'
        }`}></div>
        <h1 className={`relative text-7xl font-bold bg-gradient-to-r bg-clip-text text-transparent ${
          isDark 
            ? 'from-purple-400 via-blue-400 to-purple-400' 
            : 'from-purple-600 via-blue-600 to-purple-600'
        }`}>
          NoteMap
        </h1>
      </div>
      
      <p className={`text-4xl font-bold mb-6 ${isDark ? 'text-white' : 'text-gray-900'}`}>
        Transform Notes into Organized Knowledge
      </p>
      
      <p className={`text-xl text-center max-w-3xl mb-12 ${isDark ? 'text-gray-400' : 'text-gray-600'}`}>
        Upload, organize, and explore your notes with intelligent structure and insights
      </p>
      
      <div className="flex gap-4">
        <button
          onClick={() => setCurrentPage('upload')}
          className="px-8 py-4 bg-gradient-to-r from-purple-600 to-blue-600 text-white rounded-lg font-semibold flex items-center gap-2 hover:from-purple-700 hover:to-blue-700 transition-all duration-300 shadow-lg shadow-purple-500/50 hover:shadow-purple-500/70 hover:scale-105"
        >
          <Upload size={20} />
          Upload Notes
        </button>
        <button
          onClick={() => setCurrentPage('dashboard')}
          className={`px-8 py-4 rounded-lg font-semibold flex items-center gap-2 transition-all duration-300 hover:scale-105 ${
            isDark 
              ? 'bg-gray-800 text-gray-300 hover:bg-gray-700 border border-gray-700' 
              : 'bg-white text-gray-700 hover:bg-gray-50 border border-gray-300 shadow-md'
          }`}
        >
          <LayoutDashboard size={20} />
          View Dashboard
        </button>
      </div>
    </div>
  );

  const renderUploadPage = () => (
    <div className="min-h-screen px-6 py-12 flex flex-col items-center justify-center">
      <div className="max-w-4xl w-full mx-auto">
        <h1 className="text-6xl font-bold text-center mb-4 bg-gradient-to-r from-purple-400 via-blue-400 to-purple-400 bg-clip-text text-transparent">
          Upload Your Notes
        </h1>
        <p className={`text-lg text-center mb-16 ${isDark ? 'text-gray-400' : 'text-gray-600'}`}>
          Drop your handwritten notes and let AI organize them instantly
        </p>

        <div
          className={`rounded-2xl p-10 text-center backdrop-blur-sm transition-all duration-300 ${
            isDark 
              ? 'bg-gradient-to-br from-gray-800/40 to-gray-900/40 border' + (dragActive ? ' border-purple-500 bg-purple-500/10 border-2' : ' border-gray-700')
              : 'bg-white border' + (dragActive ? ' border-purple-500 bg-purple-100' : ' border-gray-300 shadow-lg')
          }`}
          onDragEnter={handleDrag}
          onDragLeave={handleDrag}
          onDragOver={handleDrag}
          onDrop={handleDrop}
        >
          <input
            type="file"
            ref={fileInputRef}
            multiple
            accept=".png, .jpg, .jpeg, .pdf"
            className="hidden"
            onChange={handleFileInputChange}
          />
          
          <div 
            className="cursor-pointer"
            onClick={() => fileInputRef.current?.click()}
          >
            <div className="w-20 h-20 bg-gradient-to-br from-purple-600 to-purple-700 rounded-3xl flex items-center justify-center mx-auto mb-8 shadow-xl shadow-purple-600/30">
              <Upload size={40} className="text-white" />
            </div>
            
            <h3 className={`text-2xl font-bold mb-3 ${isDark ? 'text-white' : 'text-gray-900'}`}>
              Click or drag files to upload
            </h3>
            <p className={`mb-8 text-base ${isDark ? 'text-gray-400' : 'text-gray-600'}`}>
              Supports images and PDFs up to 10MB
            </p>
            
            <div className="flex gap-3 justify-center mb-6">
              <span className={`px-5 py-2 border rounded-md text-sm font-medium ${

                                    )}
                                  </div>
                                  {(section.level || section.page) && (
                                    <div className={`flex items-center gap-2 text-xs mb-2 ${isDark ? 'text-gray-500' : 'text-gray-600'}`}>
                                      {section.level && <span>Level: H{section.level}</span>}
                                      {section.page && <span>‚Ä¢ Page: {section.page}</span>}
                                    </div>
                                  )}
                                  <p className={`text-xs leading-relaxed ${isDark ? 'text-gray-300' : 'text-gray-700'}`}>
                                    {section.desc}
                                  </p>
                                </div>
                              </div>
                            </div>
                          ))}
                          {note.sections.length > 3 && (
                            <div className={`text-xs text-center py-1 ${isDark ? 'text-gray-500' : 'text-gray-600'}`}>
                              +{note.sections.length - 3} more sections
                            </div>
                          )}
                        </div>
                      </div>
                    )}

                    {/* Key Concepts */}
                    {note.concepts && note.concepts.length > 0 && (
                      <div className={`pt-4 border-t ${isDark ? 'border-gray-700' : 'border-gray-200'}`}>
                        <p className={`text-sm font-semibold mb-3 ${isDark ? 'text-white' : 'text-gray-900'}`}>
                          Key Concepts
                        </p>
                        <div className="flex flex-wrap gap-2">
                          {note.concepts.slice(0, 6).map((concept, idx) => (
                            <span key={idx} className={`px-3 py-1.5 rounded text-xs font-medium transition-colors cursor-pointer ${
                              isDark ? 'bg-gray-700/80 text-gray-200 hover:bg-gray-600' : 'bg-gray-200 text-gray-700 hover:bg-gray-300'
                            }`}>
                              {concept}
                            </span>
                          ))}
                          {note.concepts.length > 6 && (
                            <span className={`px-3 py-1.5 rounded text-xs font-medium ${
                              isDark ? 'text-blue-400' : 'text-blue-600'
                            }`}>
                              +{note.concepts.length - 6} more
                            </span>
                          )}
                        </div>
                      </div>
                    )}
                  </div>
                </div>
              ))}
            </div>
          )}
        </div>
      </div>
    );
  };

  // Main render
  return (
    <div className={`min-h-screen transition-colors duration-300 ${
      isDark ? 'bg-gray-900' : 'bg-gradient-to-br from-gray-50 to-gray-100'
    }`}>
      {/* Navigation */}
      <nav className={`border-b px-6 py-4 ${
        isDark ? 'border-gray-800 bg-gray-900' : 'border-gray-200 bg-white shadow-sm'
      }`}>
        <div className="max-w-7xl mx-auto flex justify-between items-center">
          <div className="flex items-center gap-3">
            <div className="w-10 h-10 bg-gradient-to-br from-purple-600 to-blue-600 rounded-lg flex items-center justify-center">
              <FileText className="text-white" size={24} />
            </div>
            <span className={`text-2xl font-bold ${isDark ? 'text-white' : 'text-gray-900'}`}>
              NoteMap
            </span>
          </div>

          <div className="flex items-center gap-3">
            <button
              onClick={() => setCurrentPage('home')}
              className={`px-4 py-2 rounded-lg flex items-center gap-2 transition-all duration-300 ${
                currentPage === 'home'
                  ? 'bg-purple-600 text-white shadow-lg shadow-purple-600/50'
                  : isDark ? 'text-gray-300 hover:bg-gray-800 hover:scale-105' : 'text-gray-700 hover:bg-gray-100 hover:scale-105'
              }`}
            >
              <Home size={18} />
              Home
            </button>
            <button
              onClick={() => setCurrentPage('upload')}
              className={`px-4 py-2 rounded-lg flex items-center gap-2 transition-all duration-300 ${
                currentPage === 'upload'
                  ? 'bg-purple-600 text-white shadow-lg shadow-purple-600/50'
                  : isDark ? 'text-gray-300 hover:bg-gray-800 hover:scale-105' : 'text-gray-700 hover:bg-gray-100 hover:scale-105'
              }`}
            >
              <Upload size={18} />
              Upload
            </button>
            <button
              onClick={() => setCurrentPage('dashboard')}
              className={`px-4 py-2 rounded-lg flex items-center gap-2 transition-all duration-300 ${
                currentPage === 'dashboard'
                  ? 'bg-purple-600 text-white shadow-lg shadow-purple-600/50'
                  : isDark ? 'text-gray-300 hover:bg-gray-800 hover:scale-105' : 'text-gray-700 hover:bg-gray-100 hover:scale-105'
              }`}
            >
              <LayoutDashboard size={18} />
              Dashboard
            </button>
            <button
              onClick={() => setIsDark(!isDark)}
              className={`p-2 rounded-lg transition-all duration-300 hover:scale-105 ${
                isDark ? 'text-gray-300 hover:bg-gray-800' : 'text-gray-700 hover:bg-gray-100'
              }`}
            >
              {isDark ? <Sun size={20} /> : <Moon size={20} />}
            </button>
            <button
              onClick={() => setShowSettings(!showSettings)}
              className={`p-2 rounded-lg transition-all duration-300 hover:scale-105 ${
                isDark ? 'text-gray-300 hover:bg-gray-800' : 'text-gray-700 hover:bg-gray-100'
              }`}
            >
              <Settings size={20} />
            </button>
          </div>
        </div>
      </nav>

      {/* Settings Panel */}
      {showSettings && (
        <div className={`border-b px-6 py-4 ${
          isDark ? 'border-gray-800 bg-gray-800' : 'border-gray-200 bg-gray-50'
        }`}>
          <div className="max-w-7xl mx-auto">
            <h3 className={`text-lg font-semibold mb-3 ${isDark ? 'text-white' : 'text-gray-900'}`}>
              Backend Settings
            </h3>
            
            {backendInfo && (
              <div className={`mb-4 p-4 rounded-lg ${isDark ? 'bg-gray-900/50' : 'bg-white'}`}>
                <div className="flex items-center gap-2 mb-2">
                  <div className={`w-2 h-2 rounded-full ${
                    backendInfo.status === 'healthy' ? 'bg-green-500' : 'bg-red-500'
                  }`}></div>
                  <span className={`text-sm font-medium ${isDark ? 'text-white' : 'text-gray-900'}`}>
                    Status: {backendInfo.status === 'healthy' ? '‚úì Online' : '‚úó Offline'}
                  </span>
                </div>
                <p className={`text-xs mb-2 ${isDark ? 'text-gray-400' : 'text-gray-600'}`}>
                  {backendInfo.message}
                </p>
                {backendInfo.features && (
                  <div className="flex flex-wrap gap-1 mt-2">
                    {backendInfo.features.map((feature, idx) => (
                      <span key={idx} className={`text-xs px-2 py-1 rounded ${
                        isDark ? 'bg-purple-900/30 text-purple-300' : 'bg-purple-100 text-purple-700'
                      }`}>
                        {feature}
                      </span>
                    ))}
                  </div>
                )}
              </div>
            )}
            
            <div className="flex gap-3 items-center">
              <label className={`text-sm ${isDark ? 'text-gray-400' : 'text-gray-600'}`}>
                API Endpoint:
              </label>
              <input
                type="text"
                value={apiEndpoint}
                onChange={(e) => setApiEndpoint(e.target.value)}
                placeholder="http://localhost:5000/api/upload"
                className={`flex-1 max-w-xl rounded-lg px-4 py-2 focus:outline-none ${
                  isDark 
                    ? 'bg-gray-900 border border-gray-700 text-white focus:border-purple-500' 
                    : 'bg-white border border-gray-300 text-gray-900 focus:border-purple-500'
                }`}
              />
              <button
                onClick={checkBackendHealth}
                className="px-4 py-2 bg-purple-600 text-white rounded-lg font-medium hover:bg-purple-700 transition-all"
              >
                Test Connection
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Page Content */}
      {currentPage === 'home' && renderHomePage()}
      {currentPage === 'upload' && renderUploadPage()}
      {currentPage === 'dashboard' && renderDashboard()}

      {/* Preview Modal */}
      {showPreview && previewFile && (
        <div 
          className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-black/80 backdrop-blur-sm"
          onClick={closePreview}
        >
          <div 
            className={`relative max-w-6xl w-full max-h-[90vh] rounded-2xl overflow-hidden shadow-2xl ${
              isDark ? 'bg-gray-900' : 'bg-white'
            }`}
            onClick={(e) => e.stopPropagation()}
          >
            <div className={`flex items-center justify-between p-4 border-b ${
              isDark ? 'border-gray-800' : 'border-gray-200'
            }`}>
              <div className="flex items-center gap-3">
                <FileText className={isDark ? 'text-purple-400' : 'text-purple-600'} size={24} />
                <div>
                  <h3 className={`font-bold ${isDark ? 'text-white' : 'text-gray-900'}`}>
                    {previewFile.name}
                  </h3>
                  <p className={`text-sm ${isDark ? 'text-gray-400' : 'text-gray-600'}`}>
                    {previewFile.size}
                  </p>
                </div>
              </div>
              <button
                onClick={closePreview}
                className={`p-2 rounded-lg transition-all hover:scale-110 ${
                  isDark ? 'hover:bg-gray-800 text-gray-400 hover:text-white' : 'hover:bg-gray-200 text-gray-600 hover:text-gray-900'
                }`}
              >
                <X size={24} />
              </button>
            </div>

            <div className={`p-6 overflow-auto max-h-[calc(90vh-80px)] ${
              isDark ? 'bg-gray-800' : 'bg-gray-50'
            }`}>
              {previewFile.type && previewFile.type.startsWith('image/') ? (
                <div className="flex items-center justify-center">
                  <img 
                    src={previewFile.previewUrl} 
                    alt={previewFile.name}
                    className="max-w-full h-auto rounded-lg shadow-xl"
                  />
                </div>
              ) : previewFile.type === 'application/pdf' ? (
                <div className="w-full h-[70vh]">
                  <iframe
                    src={previewFile.previewUrl}
                    className="w-full h-full rounded-lg"
                    title={previewFile.name}
                  />
                </div>
              ) : (
                <div className={`text-center py-12 ${isDark ? 'text-gray-400' : 'text-gray-600'}`}>
                  <FileText size={64} className="mx-auto mb-4 opacity-50" />
                  <p>Preview not available for this file type</p>
                </div>
              )}
            </div>

            <div className={`flex items-center justify-between p-4 border-t ${
              isDark ? 'border-gray-800 bg-gray-900' : 'border-gray-200 bg-gray-50'
            }`}>
              <div className={`text-sm ${isDark ? 'text-gray-400' : 'text-gray-600'}`}>
                {previewFile.type && previewFile.type.startsWith('image/') ? 'Image Preview' : 'PDF Preview'}
              </div>
              <button
                onClick={() => {
                  const link = document.createElement('a');
                  link.href = previewFile.previewUrl;
                  link.download = previewFile.name;
                  link.click();
                }}
                className="px-4 py-2 bg-purple-600 text-white rounded-lg font-semibold hover:bg-purple-700 transition-all flex items-center gap-2"
              >
                <Download size={16} />
                Download
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
